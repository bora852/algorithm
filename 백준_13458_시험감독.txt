import java.util.Scanner;

public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int N = sc.nextInt();
		int[] room = new int[N];
		
		for(int i = 0; i < N; i++) {
			room[i] = sc.nextInt();
		}
		
		int B = sc.nextInt();
		int C = sc.nextInt();
		
		long ans = 0;
		for(int i = 0; i < N; i++) {
			room[i] -= B;
			ans++;
			
			if(room[i] > 0) {
				ans += Math.ceil((1.0 * room[i])/C);
			}
		}
		
		System.out.println(ans);
	}
}
